CPU Scheduling Algorithms — C Programs
1. FCFS (First Come First Serve)
#include<stdio.h>
int main(){
  int bt[10],wt[10],p[10],i,j,n,total=0;
  float avg;
  printf("Enter number of processes: ");
  scanf("%d",&n);
  for(i=1;i<=n;i++){
    printf("BT for P%d: ",i);
    scanf("%d",&bt[i]); p[i]=i;
  }
  wt[1]=0;
  for(i=2;i<=n;i++){
    wt[i]=0;
    for(j=1;j<i;j++) wt[i]+=bt[j];
    total+=wt[i];
  }
  printf("P\tBT\tWT\n");
  for(i=1;i<=n;i++)
    printf("P%d\t%d\t%d\n",p[i],bt[i],wt[i]);
  avg=(float)total/n;
  printf("Total WT: %d\nAvg WT: %.2f\n",total,avg);
}
WTᵢ = BT₁ + BT₂ + ... + BTᵢ₋₁ · No sorting needed · Arrival order = execution order
2. SJF (Shortest Job First)
#include<stdio.h>
int main(){
  int bt[10],p[10],wt[10],temp,i,j,n,total=0;
  float avg;
  printf("Enter total number of processes: ");
  scanf("%d",&n);
  for(i=0;i<n;i++){
    printf("BT for P%d: ",i);
    scanf("%d",&bt[i]); p[i]=i;
  }
  /* Bubble sort by burst time (ascending) */
  for(i=0;i<n-1;i++)
    for(j=0;j<n-i-1;j++)
      if(bt[j]>bt[j+1]){
        temp=bt[j];bt[j]=bt[j+1];bt[j+1]=temp;
        temp=p[j];p[j]=p[j+1];p[j+1]=temp;
      }
  wt[0]=0;
  for(i=1;i<n;i++){
    wt[i]=0;
    for(j=0;j<i;j++) wt[i]+=bt[j];
    total+=wt[i];
  }
  printf("P\tBT\tWT\n");
  for(i=0;i<n;i++)
    printf("P%d\t%d\t%d\n",p[i],bt[i],wt[i]);
  avg=(float)total/n;
  printf("Total WT: %d\nAvg WT: %.2f\n",total,avg);
}
Sort by BT ascending → then calculate WT same as FCFS · Gives minimum average WT (optimal)
3. Priority Scheduling
#include<stdio.h>
int main(){
  int bt[10],pr[10],wt[10],p[10],i,temp,j,n,total=0;
  float avg;
  printf("Enter number of processes: ");
  scanf("%d",&n);
  for(i=0;i<n;i++){
    printf("BT & Priority for P%d: ",i);
    scanf("%d %d",&bt[i],&pr[i]); p[i]=i;
  }
  /* Bubble sort by priority (ascending = lower num = higher prio) */
  for(i=0;i<n-1;i++)
    for(j=0;j<n-i-1;j++)
      if(pr[j]>pr[j+1]){
        temp=pr[j];pr[j]=pr[j+1];pr[j+1]=temp;
        temp=bt[j];bt[j]=bt[j+1];bt[j+1]=temp;
        temp=p[j];p[j]=p[j+1];p[j+1]=temp;
      }
  wt[0]=0;
  for(i=1;i<n;i++){
    wt[i]=0;
    for(j=0;j<i;j++) wt[i]+=bt[j];
    total+=wt[i];
  }
  printf("P\tBT\tWT\n");
  for(i=0;i<n;i++)
    printf("P%d\t%d\t%d\n",p[i],bt[i],wt[i]);
  avg=(float)total/n;
  printf("Total WT: %d\nAvg WT: %.2f\n",total,avg);
}
Lower priority number = runs first · Sort by pr[] ascending · Same WT formula as FCFS/SJF after sorting
4. Round Robin (with Time Quantum)
#include<stdio.h>
int main(){
  int bt[10],wt[10],rt[10],p[10],ct[10],done,i,n,tq,time=0,total=0;
  float avg;
  printf("Enter number of processes: "); scanf("%d",&n);
  for(i=1;i<=n;i++){
    printf("BT for P%d: ",i); scanf("%d",&bt[i]);
    rt[i]=bt[i]; p[i]=i;         // rt = remaining time
  }
  printf("Enter time quantum: "); scanf("%d",&tq);
  while(1){
    done=1;
    for(i=1;i<=n;i++){
      if(rt[i]>0){
        done=0;
        if(rt[i]>tq){ time+=tq; rt[i]-=tq; }
        else{ time+=rt[i]; wt[i]=time-bt[i]; ct[i]=time; rt[i]=0; }
        printf("P%d ",i);
      }
    }
    if(done==1) break;
  }
  printf("\nP\tBT\tWT\n");
  for(i=1;i<=n;i++){ printf("P%d\t%d\t%d\n",p[i],bt[i],wt[i]); total+=wt[i]; }
  avg=(float)total/n;
  printf("Total WT: %d\nAvg WT: %.2f\n",total,avg);
}
rt[] = remaining time copy of bt[]. WT = completion time − burst time. Each process gets tq units per round. Loops until all rt[] = 0.
📊 Algorithm Comparison
AlgorithmKey IdeaPreemptive?
FCFSArrival order, no sortingNo
SJFSort by BT ascendingNo (here)
PrioritySort by PR ascendingNo (here)
Round RobinTime quantum, circularYes
Formulas
Waiting Time (WT):
WT[0] = 0
WT[i] = WT[i-1] + BT[i-1]

Turnaround Time (TAT):
TAT[i] = WT[i] + BT[i]

Average WT:
Avg = (float)total / n

Round Robin WT:
WT[i] = completion_time - BT[i]
⚠ Arrays in FCFS start at index 1. SJF/Priority start at 0. Don't mix them!
